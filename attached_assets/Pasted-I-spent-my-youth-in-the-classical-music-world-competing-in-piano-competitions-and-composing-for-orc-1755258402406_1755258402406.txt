<ul id="youtube-list" class="essay-list"></ul>
<script>
  (async () => {
    try {
      const res = await fetch('/assets/data/youtube.json');
      const items = await res.json();
      const ul = document.getElementById('youtube-list');
      items.forEach(item => {
        const d = new Date(item.date);
        const mm = d.getMonth() + 1, dd = d.getDate(), yy = String(d.getFullYear()).slice(-2);
        const li = document.createElement('li');
        const a = document.createElement('a');
        a.href = item.link; a.target = '_blank'; a.rel = 'noopener';
        a.textContent = `${item.title}  –  ${mm}/${dd}/${yy}`;
        li.appendChild(a); ul.appendChild(li);
      });
    } catch {}
  })();
</script>
